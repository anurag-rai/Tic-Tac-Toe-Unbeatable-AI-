comp_move() {            //MAIN MOVE FUNCTION OF THE COMPUTER
    check_win();         //checks if there a move that wins the game for AI
    check_oppo_win();    //checks if there a move that wins the game for Player, and blocks it
    create_fork();       //checks if a fork can be created
    block_fork();        //checks if an opponent can fork, and stop it
    occupy_centre();     //if centre if unoccupied, take it
    occupy_corner();     //if any corner if unoccupied, take it
    occupy_edge();       //if any edge if unoccupied, take it
}
